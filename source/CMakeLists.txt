# contrib folders
INCLUDE_DIRECTORIES(${OCTOON_PATH_DEPENDENCIES}/zlib)
INCLUDE_DIRECTORIES(${OCTOON_PATH_DEPENDENCIES}/glew)
INCLUDE_DIRECTORIES(${OCTOON_PATH_DEPENDENCIES}/glfw/include)
INCLUDE_DIRECTORIES(${OCTOON_PATH_DEPENDENCIES}/imgui)

# octoon folder
INCLUDE_DIRECTORIES(${OCTOON_PATH_ITNCLUDE})

# lib folder for link
LINK_DIRECTORIES(${OCTOON_LIBRARY_OUTPUT_PATH})

OPTION(OCTOON_FEATURE_UI_ENABLE "On for enable off for disable" ON)
OPTION(OCTOON_FEATURE_INPUT_ENABLE "On for enable off for disable" ON)

# runtime libraries
ADD_SUBDIRECTORY(octoon-math)
ADD_SUBDIRECTORY(octoon-runtime)
ADD_SUBDIRECTORY(octoon-graphics)
ADD_SUBDIRECTORY(octoon-io)

IF(OCTOON_FEATURE_INPUT_ENABLE)
	ADD_DEFINITIONS(-DOCTOON_FEATURE_INPUT_ENABLE)
	ADD_SUBDIRECTORY(octoon-input)
ENDIF()

IF(OCTOON_FEATURE_INPUT_ENABLE)
	ADD_DEFINITIONS(-DOCTOON_FEATURE_UI_ENABLE)
	ADD_SUBDIRECTORY(octoon-ui)
ENDIF()

# framework
ADD_SUBDIRECTORY(octoon)
ADD_SUBDIRECTORY(octoon-c)
